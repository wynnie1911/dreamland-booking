// ============================================================ // MY LITTLE DREAMLAND — Booking API (Apps Script) // ============================================================ // Deploy: Extensions > Apps Script > Deploy > New deployment // Type: Web App | Execute as: Me | Access: Anyone // ============================================================ const CONFIG = { SHEET_ID: "1lRXe8Tt02mNZ33u6wIaoWNbxOz2CtAlwTbeGIemVnFE", // ID Sheet của bạn OPEN_TIME: "08:30", // Giờ mở cửa CLOSE_TIME: "20:00", // Giờ đóng cửa (slot cuối kết thúc lúc này) SLOT_DURATION: 60, // Phút mỗi slot MAX_CAPACITY: 40, // Tối đa trẻ mỗi slot SHEET_BOOKINGS: "Bookings", SHEET_CAPACITY: "Capacity", }; // ============================================================ // ROUTING — GET / POST // ============================================================ function doGet(e) { const action = e.parameter.action || "getSlots"; if (action === "getSlots") { const date = e.parameter.date; if (!date) return jsonError("Thiếu tham số: date"); return jsonResponse(getSlotsForDate(date)); } if (action === "createBooking") { const data = { action: "createBooking", date: e.parameter.date, slot: e.parameter.slot, parentName: e.parameter.parentName, parentEmail: e.parameter.parentEmail, parentPhone: e.parameter.parentPhone || "", numChildren: e.parameter.numChildren, notes: e.parameter.notes || "", }; return jsonResponse(createBooking(data)); } return jsonError("Action không hợp lệ"); } function doPost(e) { try { let body; if (e.parameter && e.parameter.data) { body = JSON.parse(e.parameter.data); } else if (e.postData && e.postData.contents) { body = JSON.parse(e.postData.contents); } else { return jsonError("Không đọc được request body"); } const action = body.action || "createBooking"; if (action === "createBooking") { return jsonResponse(createBooking(body)); } return jsonError("Action không hợp lệ"); } catch (err) { return jsonError("Request không hợp lệ: " + err.message); } } // ============================================================ // GET SLOTS — Trả về danh sách slot + số chỗ còn lại // ============================================================ function getSlotsForDate(date) { const slots = generateSlots(); const booked = getBookedCountsByDate(date); const customCapacity = getCustomCapacity(date); // Lấy thời gian hiện tại theo đúng múi giờ Melbourne (Úc) để check slot quá khứ chuẩn xác const now = new Date(); const todayStr = Utilities.formatDate(now, "Australia/Melbourne", "yyyy-MM-dd"); const currentMinutes = parseInt(Utilities.formatDate(now, "Australia/Melbourne", "HH")) * 60 + parseInt(Utilities.formatDate(now, "Australia/Melbourne", "mm")); return { date: date, slots: slots.map((slot) => { const capacity = customCapacity[slot.key] ?? CONFIG.MAX_CAPACITY; const bookedCount = booked[slot.key] || 0; const remaining = Math.max(0, capacity - bookedCount); // Ẩn slot đã qua (nếu là ngày hôm nay) let isPast = false; if (date === todayStr) { const [h, m] = slot.start.split(":").map(Number); isPast = h * 60 + m < currentMinutes; } return { key: slot.key, // "08:30-09:30" start: slot.start, // "08:30" end: slot.end, // "09:30" capacity: capacity, booked: bookedCount, remaining: remaining, status: isPast ? "past" : remaining === 0 ? "full" : remaining <= 5 ? "almostFull" : "available", }; }), }; } // ============================================================ // CREATE BOOKING // ============================================================ function createBooking(data) { // Validate input const required = ["date", "slot", "parentName", "parentEmail", "numChildren"]; for (const field of required) { if (!data[field]) return { success: false, error: `Thiếu thông tin: ${field}` }; } const numChildren = parseInt(data.numChildren); if (isNaN(numChildren) || numChildren < 1 || numChildren > 10) { return { success: false, error: "Số trẻ không hợp lệ (1-10)" }; } // Check capacity còn lại const slotsData = getSlotsForDate(data.date); const slot = slotsData.slots.find((s) => s.key === data.slot); if (!slot) return { success: false, error: "Slot không tồn tại" }; if (slot.status === "past") return { success: false, error: "Slot này đã qua" }; if (slot.remaining < numChildren) { return { success: false, error: `Không đủ chỗ. Còn lại: ${slot.remaining} chỗ`, remaining: slot.remaining, }; } // Ghi vào Sheet const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_BOOKINGS); const bookingId = generateId(); const timestamp = new Date(); sheet.appendRow([ bookingId, Utilities.formatDate(timestamp, "Australia/Melbourne", "yyyy-MM-dd HH:mm:ss"), data.date, data.slot, data.parentName, data.parentEmail, data.parentPhone || "", numChildren, data.notes || "", "confirmed", ]); // Gửi email xác nhận try { sendConfirmationEmail(data, bookingId, slot); } catch (emailErr) { console.log("Email error:", emailErr.message); } return { success: true, bookingId: bookingId, message: "Đặt vé thành công!", booking: { id: bookingId, date: data.date, slot: data.slot, numChildren: numChildren, parentName: data.parentName, }, }; } // ============================================================ // HELPERS // ============================================================ function generateSlots() { const slots = []; const [openH, openM] = CONFIG.OPEN_TIME.split(":").map(Number); const [closeH, closeM] = CONFIG.CLOSE_TIME.split(":").map(Number); const openTotal = openH * 60 + openM; const closeTotal = closeH * 60 + closeM; let current = openTotal; while (current + CONFIG.SLOT_DURATION <= closeTotal) { const start = minutesToTime(current); const end = minutesToTime(current + CONFIG.SLOT_DURATION); slots.push({ key: `${start}-${end}`, start, end }); current += CONFIG.SLOT_DURATION; } return slots; } function minutesToTime(minutes) { const h = Math.floor(minutes / 60).toString().padStart(2, "0"); const m = (minutes % 60).toString().padStart(2, "0"); return `${h}:${m}`; } function getBookedCountsByDate(date) { const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_BOOKINGS); const data = sheet.getDataRange().getValues(); const counts = {}; for (let i = 1; i < data.length; i++) { const row = data[i]; let rowDateStr = ""; if (row[2] instanceof Date) { rowDateStr = Utilities.formatDate(row[2], "Australia/Melbourne", "yyyy-MM-dd"); } else { rowDateStr = String(row[2]).trim(); } if (rowDateStr === date && row[9] === "confirmed") { const slotKey = row[3]; const numChildren = parseInt(row[7]) || 0; counts[slotKey] = (counts[slotKey] || 0) + numChildren; } } return counts; } function getCustomCapacity(date) { const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_CAPACITY); if (!sheet) return {}; const data = sheet.getDataRange().getValues(); const custom = {}; for (let i = 1; i < data.length; i++) { const row = data[i]; // ĐÃ SỬA: Định nghĩa rõ ràng dòng hiện tại trong vòng lặp const rowDate = row[0]; if (!rowDate) continue; let rowDateStr = (rowDate instanceof Date) ? Utilities.formatDate(rowDate, "Australia/Melbourne", "yyyy-MM-dd") : String(rowDate).trim(); if (rowDateStr === date && row[1] && row[2]) { custom[row[1]] = parseInt(row[2]); } } return custom; } function generateId() { return "BK" + new Date().getTime().toString(36).toUpperCase(); } function sendConfirmationEmail(data, bookingId, slot) { const subject = `Booking Confirmation - My Little Dreamland [ #${bookingId} ]`; let formattedDate = data.date; try { const d = new Date(data.date + 'T00:00:00'); formattedDate = d.toLocaleDateString('en-AU', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' }); } catch (e) {} const htmlBody = `

Booking Confirmed

Dear ${data.parentName},

Thank you for choosing My Little Dreamland. Your booking has been successfully confirmed. Please find your session details below:

Booking Reference: ${bookingId}
Date: ${formattedDate}
Session Time: ${slot.start} - ${slot.end}
Number of Children: ${data.numChildren} ${parseInt(data.numChildren) > 1 ? 'children' : 'child'}

Important Information:

• Location: 110 Garsed Street, Bendigo VIC 3550
• Please arrive 5-10 minutes prior to your scheduled session time.
• Socks are required for both adults and children in the play areas.

If you need to amend or cancel your booking, please reply directly to this email or contact us at hello@mylittledreamland.com.au.

We look forward to welcoming you and your little ones.

Warm regards,
My Little Dreamland Team

`; GmailApp.sendEmail(data.parentEmail, subject, "", { htmlBody: htmlBody }); } function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } function jsonError(message) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: message })).setMimeType(ContentService.MimeType.JSON); }
